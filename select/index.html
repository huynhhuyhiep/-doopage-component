Select

Base on https://material-ui.com/components/text-fields/

Related component

  • https://material-ui.com/components/selects/#select

Icon

  • https://material-ui.com/components/material-icons/

Basic usage

import { Select } from '@doopage/react-ui-kit'
const options = [{id: '1', name: "option 1"}, {id: '2', name: "option 2"}]
const Example = () => {
    return <Select options={options}/>;
}
export default Example;

Playground

Simple Select

option 1
() => {
  const [value, setValue] = useState('1')
  const options = [
    { id: '1', name: 'option 1' },
    { id: '2', name: 'option 2' },
  ]
  return (
    <Select
      options={options}
      label="Select"
      fullWidth
      value={value}
      onChange={e => setValue(e.target.value)}
    />
  )
}

Multiple Select

option 1
() => {
  const [value, setValue] = useState(['1'])
  const options = [
    { id: '1', name: 'option 1' },

  ]
  return (
    <Select
      multiple
      options={options}
      label="Select"
      fullWidth
      value={value}
      onChange={e => setValue(e.target.value)}
    />

}
