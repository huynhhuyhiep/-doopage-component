## Playground

### Basic
<Playground>
  {() =>
  {
    const [value, setValue] = React.useState('female');
    const handleChange = (event) =>
    {
      setValue(event.target.value);
    };
    return (
      <FormControl component="fieldset">
        <FormLabel component="legend">Gender</FormLabel>
        <RadioGroup value={value} onChange={handleChange}>
          <Radio value="female" label="Female" color='primary' />
          <Radio value="male" label="Male" color='info' />
          <Radio value="other" label="Other" disabled color='warning' />
          <Radio value="helper" label='Helper text' helperText='this is heplerText' />
        </RadioGroup>
      </FormControl>
    )
  }}
</Playground>
