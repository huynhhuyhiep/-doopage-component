# Popup
Base on
- https://material-ui.com/components/popover/#popover
- https://github.com/jcoreio/material-ui-popup-state

Icon
- https://material-ui.com/components/material-icons/

## Basic usage

```jsx
import { Popup } from '@doopage/react-ui-kit'

const Example = () => {
    return <Popup label={'menu'}>this is popup</Popup>;
}

export default Example;
```

### Control
<Playground>
  {() =>
  {
    const popupRef = useRef()
    const handleClose = () =>
    {
      popupRef.current.close()
    }
    return (
      <>
        <Popup label={<MenuIcon />} ref={popupRef}>
          <Button color='gray' simple onClick={handleClose}>
            Close popup
          </Button>
        </Popup>
      </>
    )
  }}
</Playground>
