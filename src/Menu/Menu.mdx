# Popup

## Basic usage

```jsx
import { Menu } from '@doopage/react-ui-kit'

const Example = () => {
    return (
      <Menu
        label={'menu'}
        options={[{id: '1', name: 'option1', onClick: () => {console.log('on click')}}]}
      />
    );
}

export default Example;
```
