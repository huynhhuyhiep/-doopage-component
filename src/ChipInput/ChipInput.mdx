### Advanced
<Playground>
  {() =>
  {
    const [value, setValue] = useState(['default value', 'llll'])
    return (
      <div>
        <div style={{ fontSize: 15, marginBottom: 10 }}>Value: {value.join(', ')}</div>
        <ChipInput
          allowCreate
          multiple
          placeholder='Enter to Submit'
          label='Chip Input'
          variant='outlined'
          value={value}
          onChange={(value) =>
          {
            setValue(value);
          }}
          chipProps={{
            outlined: false,
            rounded: true,
            color: 'warning'
          }}
          setCreateText={value => `Add "${value}"`}
          showCreateText={true}
        />
      </div>
    )
  }}
</Playground>
