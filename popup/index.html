Popup

Basic usage

import { Popup } from '@doopage/react-ui-kit'
const Example = () => {
    return <Popup label={'menu'}>this is popup</Popup>;
}
export default Example;

Advanced

<Popup
  label={<MenuIcon />}
  buttonProps={{ justIcon: true, color: 'gray', size: 'small', round: true }}
  anchorOrigin={{
    vertical: 'bottom',
    horizontal: 'left',
  }}
  transformOrigin={{
    vertical: 'top',
    horizontal: 'left',
  }}
>
  <div>this is pop up</div>
</Popup>

Control

() => {
  const popupRef = useRef()
  const handleClose = () => {
    popupRef.current.close()
  }
  return (
    <>
      <Popup label={<MenuIcon />} ref={popupRef}>
        <Button color="gray" simple onClick={handleClose}>
          Close popup
        </Button>
      </Popup>
    </>
  )
}
