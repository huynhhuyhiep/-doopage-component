Accordion

Base on https://material-ui.com/components/accordion/

Basic usage

import { Accordion } from '@doopage/react-ui-kit'


const Example = () => <Accordion title='title'>content</Accordion>;


export default Example;

Advance

This is title with icon
() => {
  const [expand, setExpand] = useState(false)
  return (
    <Card>
      <Accordion
        expanded={expand}
        title={
          <>
            <InsertCommentIcon style={{ marginRight: 5 }} />
            This is title with icon
          </>
        }
        expandIcon={<Switch checked={expand} />}
        rotateExpandIcon={false}
        onChange={() => setExpand(prevState => !prevState)}
      >
        <Input label="this is input" />
        <Button>Button</Button>
      </Accordion>
    </Card>
  )
}
